function calcSc(id){

function setupDrag(idx){
  const fw=document.getElementById('fw'+idx);if(!fw)return;
  const svg=fw.querySelector('svg');if(!svg)return;
  svg.querySelectorAll('.pd').forEach(dot=>{
    let dr=false,ox=0,oy=0;
    function getXY(e){
      const t=e.touches?e.touches[0]:e;
      const r=svg.getBoundingClientRect();
      const vb=svg.viewBox.baseVal;
      const scaleX=vb.width/r.width;
      const scaleY=vb.height/r.height;
      return{
        x:(t.clientX-r.left)*scaleX,
        y:(t.clientY-r.top)*scaleY
      };
    }
    function onDown(e){
      dr=true;
      const p=getXY(e);
      ox=p.x-parseFloat(dot.getAttribute('cx'));
      oy=p.y-parseFloat(dot.getAttribute('cy'));
      e.preventDefault();
      e.stopPropagation();
    }
    function onMove(e){
      if(!dr)return;
      e.preventDefault();
      const p=getXY(e);
      const nx=Math.max(10,Math.min(270,p.x-ox));
      const ny=Math.max(10,Math.min(170,p.y-oy));
      dot.setAttribute('cx',nx);
      dot.setAttribute('cy',ny);
      const tx=dot.nextElementSibling;
      if(tx&&tx.tagName==='text'){
        tx.setAttribute('x',nx);
        tx.setAttribute('y',ny+3);
      }
    }
    function onUp(){dr=false;}
    dot.addEventListener('mousedown',onDown,{passive:false});
    dot.addEventListener('touchstart',onDown,{passive:false});
    svg.addEventListener('mousemove',onMove,{passive:false});
    svg.addEventListener('touchmove',onMove,{passive:false});
    svg.addEventListener('mouseup',onUp);
    svg.addEventListener('touchend',onUp);
    document.addEventListener('mouseup',onUp);
    document.addEventListener('touchend',onUp);
  });
}

function doImp(inp){
  const file=inp.files[0];if(!file)return;
  const reader=new FileReader();
  reader.onload=e=>{
    try{
      const imported=JSON.parse(e.target.result);
      // تحقق من صحة الملف
      if(!imported.classes||!imported.sessions){toast('❌ ملف غير صالح');return;}
      // احتفظ بمفتاح API الحالي
      const currentKey=DB.apiKey;
      DB={...imported,apiKey:currentKey};

    }catch(err){toast('❌ خطأ في قراءة الملف');}
  };
  reader.readAsText(file);
  inp.value='';
}
